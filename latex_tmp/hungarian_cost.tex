\section*{Setup}
At frame $i$ (dataset id $\text{fid}$) we have
\begin{itemize}
  \item $N$ OWL detections $d_1,\ldots,d_N$, each with bbox $b_d \in \mathbb{R}^4$,
        label $\ell_d$, score $s_d \in [0,1]$;
  \item $M$ track entries $t_1,\ldots,t_M$, each with object id $\text{oid}_t$, label
        history $\mathcal{L}_t \subseteq \Sigma$ (the set of labels the track has ever been
        observed under), and a reference mask/bbox:
        \begin{itemize}
            \item \emph{active} track: binary mask $m_t \in \{0,1\}^{H\times W}$ from
                  the current SAM2 propagation;
            \item \emph{dormant} track (no current mask, but seen within the last
                  30 client frames): last-valid bbox $\hat b_t \in \mathbb{R}^4$.
        \end{itemize}
\end{itemize}

\section*{Pairwise cost}
Define the IoU overlap
\begin{equation}
    o(d, t) \;=\;
    \begin{cases}
        \mathrm{IoU}_{\text{mask}}(b_d, m_t) & \text{if $t$ is active} \\[2pt]
        \mathrm{IoU}_{\text{bbox}}(b_d, \hat b_t) & \text{if $t$ is dormant} \\[2pt]
        0 & \text{otherwise}
    \end{cases}
\end{equation}
where (with $R(b)$ denoting the clipped rectangle of bbox $b$, and $|\cdot|$
pixel count)
\begin{align}
    \mathrm{IoU}_{\text{mask}}(b_d, m_t) &=
        \frac{\big|\,m_t \cap R(b_d)\,\big|}
             {|R(b_d)| + |m_t| - |\,m_t \cap R(b_d)\,|}, \\[4pt]
    \mathrm{IoU}_{\text{bbox}}(b_d, \hat b_t) &=
        \frac{|R(b_d) \cap R(\hat b_t)|}{|R(b_d) \cup R(\hat b_t)|}.
\end{align}

The soft label penalty is
\begin{equation}
    p_{\text{lbl}}(d, t) \;=\;
    \begin{cases}
        0 & \ell_d \in \mathcal{L}_t \\
        \lambda_{\text{lbl}} & \ell_d \notin \mathcal{L}_t
    \end{cases}
    \qquad \text{(default } \lambda_{\text{lbl}} = 0.2 \text{)}.
\end{equation}

The soft confidence penalty is
\begin{equation}
    p_{\text{cnf}}(d) \;=\; \lambda_{\text{cnf}} \,\bigl(1 - s_d\bigr)
    \qquad \text{(default } \lambda_{\text{cnf}} = 0.15 \text{)}.
\end{equation}
It deprioritises low-confidence OWL detections during matching. The
weight is chosen smaller than $\lambda_{\text{lbl}}$ so the score term
nudges Hungarian's ordering without vetoing legitimate matches to
drifted tracks.

The assignment cost is
\begin{equation}
    \boxed{\;
        C(d, t) \;=\;
        \begin{cases}
            (1 - o(d,t)) + p_{\text{lbl}}(d,t) + p_{\text{cnf}}(d)
                & \text{if } o(d,t) > 0 \\
            +\infty & \text{otherwise}
        \end{cases}
    \;}
\end{equation}
so $C(d,t) \in [0,\,1 + \lambda_{\text{lbl}} + \lambda_{\text{cnf}}]
\cup \{+\infty\}$.

\section*{Hungarian step}
Build the cost matrix $\mathbf{C} \in \mathbb{R}^{N \times M}$ with entries
$C_{ij} = C(d_i, t_j)$, pad missing / invalid entries with a large sentinel.
Solve
\begin{equation}
    \sigma^\star \;=\; \arg\min_{\sigma}
        \sum_{i \in \text{matched}(\sigma)} C_{i,\sigma(i)}
\end{equation}
via \texttt{scipy.optimize.linear\_sum\_assignment}. Each optimal pair
$(i,\sigma^\star(i))$ is \emph{accepted} iff
\begin{equation}
    C_{i,\sigma^\star(i)} \;\le\; \tau, \qquad \text{(default } \tau = 0.7 \text{)}.
\end{equation}
Accepted pairs update the track with the OWL observation:
$t_{\sigma^\star(i)}\text{.label\_scores}[\ell_{d_i}][i] = s_{d_i}$.

\section*{Single-direction fallback (merge unmatched)}
Let $U \subseteq \{1,\ldots,N\}$ be the set of detections that Hungarian did
\emph{not} accept (either unassigned, or assigned with $C_{i,\sigma^\star(i)}
> \tau$). Process $i \in U$ sequentially; at each step the current track set
$\{t_1,\ldots,t_M\}$ includes tracks newly seeded earlier in the same
frame's fallback loop. For each $i$,
\begin{equation}
    j^\star \;=\; \arg\min_{j \in \{1,\ldots,M\}} C_{ij}.
\end{equation}
\textbf{If} $C_{i,j^\star} \le \tau$: absorb $d_i$ into $t_{j^\star}$ (same
update rule as above). Multiple unmatched detections may be absorbed into
the same track.\\
\textbf{Else if} $s_{d_i} \ge s_{\min}$ (default $s_{\min} = 0.25$): seed a
new track $t_{M+1}$ from $d_i$'s bbox, insert it into the track set.\\
\textbf{Else} (low-confidence, no plausible existing match): \emph{drop}
$d_i$ -- do not create a track.

\section*{Reading off the defaults ($\tau = 0.7$, $\lambda_{\text{lbl}} = 0.2$,
$\lambda_{\text{cnf}} = 0.15$)}
The acceptance condition $C \le \tau$ can be rewritten as a lower bound on
IoU:
\begin{equation}
    o(d, t) \;\ge\; 1 - \tau + p_{\text{lbl}}(d, t) + p_{\text{cnf}}(d).
\end{equation}
\begin{itemize}
    \item Matching label ($p_{\text{lbl}} = 0$) and a high-score detection
          ($s_d = 0.9$): required IoU $\ge 1 - 0.7 + 0 + 0.015 \approx 0.32$.
    \item Matching label, low-score detection ($s_d = 0.2$):
          required IoU $\ge 1 - 0.7 + 0 + 0.12 = 0.42$.
    \item Mismatched label, $s_d = 0.5$: required IoU $\ge 1 - 0.7 + 0.2 +
          0.075 \approx 0.58$.
\end{itemize}
So a mismatched-label pair needs roughly $\lambda_{\text{lbl}}$ extra IoU,
and a low-score detection needs roughly $\lambda_{\text{cnf}} (1 - s_d)$
extra IoU, to beat the threshold.

\section*{What is \emph{not} in the pipeline}
There is no post-propagation mask NMS between tracks, no cross-track
absorption or mask blending, and no label-flip between tracks. Tracks are
only created by the fallback (step 4) and only destroyed by the periodic
session reset (when their EMA mask is empty). Cross-track decisions live
entirely in the Hungarian / fallback layer above.
