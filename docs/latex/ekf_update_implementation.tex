\section{Scope}
%-----------------------------------------------------------------------------
Implementation map for one frame of \code{TwoTierOrchestrator.step()} in
\code{pose\_update/orchestrator.py}, with \code{B-eq:*} cross-references
into \code{bernoulli\_ekf.tex} (the formulation paper, authoritative on
ambiguity). Per-track state lives in \emph{base frame}: $X^{(i)}=
(\ell,T_{bo},P_{bo},r,\tau,\mathcal H)$. SLAM covariance $\Sigma_{wb}$
never enters the recursion (it appears only in the world-frame emit).

%-----------------------------------------------------------------------------
\section{Pipeline overview}
%-----------------------------------------------------------------------------

Standard MOT order, predict-then-measure:
\begin{equation*}
\begin{aligned}
   & (T_{wb,k},T_{bc}(t_k),T_{bg,k},\,\text{phase}_k,\,
      \{m_l,\hat\ell_l,\hat s_l,\tau_l\}_l) \\
   &\quad\xrightarrow{\textbf{0/1}\;\text{ingest+predict}}\;
      (T,P)_{k|k-1}
   \;\xrightarrow{\textbf{2}\;\text{centroid (cleaned mask)}}\;
      \{c^{co}_l\} \\
   &\quad\xrightarrow{\textbf{3}\;\text{Hungarian on }d^2_t}\;
      \pi^*
   \;\xrightarrow{\textbf{3b}\;\text{ICP at filter prior}}\;
      \{(\hat T^{co}_l,\hat R^{co}_l)\}_{l\in\pi^*} \\
   &\quad\xrightarrow{\textbf{4-6}\;\text{vis+update+miss}}\;
      X_k
   \;\xrightarrow{\textbf{7}\;\text{pending+admit}}\;
      X_k\cup X^{\text{new}}
   \;\xrightarrow{\textbf{8}\;\text{prune+merge}}\;
      \widehat{\mathcal X}_k
\end{aligned}
\end{equation*}

\paragraph{Two non-trivial design constraints.}
\begin{itemize}[leftmargin=1.5em,itemsep=0pt,topsep=2pt]
\item \textbf{Tracker oid is state-only.} The detection's \code{id}
   (perception's per-mask tracking id, e.g.\ SAM2's \code{tau}) is
   metadata, never tracker identity. Tracker oids are minted by
   \code{\_next\_track\_id()} (fresh consecutive int) at admission only.
\item \textbf{ICP is a measurement, not an admission gate.} For matched
   pairs ICP refines $(T,R)$; for births ICP runs \emph{exactly once
   on admission} (centroid-anchored, populates
   \code{PoseEstimator.\_refs[oid]}). The pre-2026 ``ICP probes every
   unmatched detection then admit'' path leaked stale reference clouds
   and is removed.
\end{itemize}

\paragraph{Per-frame algorithm (one line each).}
\begin{description}[leftmargin=2em,style=nextline,itemsep=2pt,topsep=2pt]
\item[\textbf{0.}] Ingest $T_{wb,k}$, install $T_{bc}(t_k)$
   (\S\ref{sec:Tbc}); cache $T_{bg,k}$ for held tracks.
\item[\textbf{1.}] Predict: static tracks via $u_k=T_{wb,k}^{-1}T_{wb,k-1}$;
   held tracks via $\Delta T_{bg}$ with phase-dependent $Q$
   (\S\ref{sec:held}); $r\!\leftarrow\!p_s r$.
\item[\textbf{2.}] Per-mask cleaned-cloud centroid $c^{co}_l\in\R^3$.
\item[\textbf{3.}] Hungarian on the 3-DOF $d^2_t$, $\chi^2_3(0.9997)=21.108$ gate.
\item[\textbf{3b.}] One ICP per matched pair, seeded with
   $\hat T^{co}_{k|k-1,i}=T_{cb}(t_k)T_{bo,k|k-1}^{(i)}$ against
   \code{\_refs[oid]}.
\item[\textbf{4.}] Visibility $p_v^{(i)}$.
\item[\textbf{5.}] Matched update: IEKF + Joseph + Huber on $d^2_t$;
   $r$ via \code{B-eq:r\_assoc}; chain append.
\item[\textbf{6.}] Missed: $r$ via \code{B-eq:r\_miss}; $T,P$ unchanged.
\item[\textbf{7.}] \textbf{Birth via pending buffer}:
   policy gate (border + tracker-side $n_{\text{obs}}\!\ge\!K$ + score)
   $\to$ if admitted, \code{\_next\_track\_id()} + ICP once + commit
   (\S\ref{sec:birth}).
\item[\textbf{8.}] Prune $r<r_{\min}$; greedy self-merge with
   held-id-as-keeper (\S\ref{sec:held}).
\item[\textbf{9.}] Emit $\{i:r\!\ge\!r_{\text{conf}}\}$; world-frame
   composition $T_{wo}\!=\!T_{wb}T_{bo}$.
\end{description}

%-----------------------------------------------------------------------------
\section{Per-frame steps}
%-----------------------------------------------------------------------------

\subsection{Step 0. Ingest + per-frame camera extrinsic}\label{sec:Tbc}

\code{ingest\_slam} caches $T_{wb,k},T_{wb,k-1}$ for the predict's body-form
control $u_k=T_{wb,k}^{-1}T_{wb,k-1}$. The world-form alternative
$T_{wb,k-1}T_{wb,k}^{-1}$ produces $\sim$1\,m drift on real trajectories
(adjoint-conjugation mismatch).

The base-to-camera-optical extrinsic $T_{bc}(t_k)$ is \emph{time-varying}
on Fetch (head pan/tilt + torso lift). The driver loads it from
\code{pose\_txt/T\_bc.txt} (one row per frame) and installs it via
\code{set\_camera\_extrinsic} \emph{before} any measurement-side call.
Treating $T_{bc}$ as constant aliases head motion as object motion and
silently inflates $d^2_t$ on every panned frame. Static-predict $u_k$
is $T_{bc}$-free; only the measurement lift and the ICP prior depend
on it.

\subsection{Step 1. Predict (state + existence)}\label{sec:predict}

\textbf{Static tracks} (the default, \code{predict\_static} with
\code{skip\_oids=held\_oids}): mean $\mu\!\leftarrow\!u_k\mu$,
$P\!\leftarrow\!\Phi(\Ad{u_k}P\Ad{u_k}^{\!\top}+Q)$, then per-axis
floor (\code{cfg.P\_min\_diag}, defaults $5\,\text{mm}/0.05\,\text{rad}$).
The floor is not cosmetic: without it, $P$ shrinks below per-frame
perception jitter after $N\!\gg\!1$ updates and the $\chi^2_3$ gate
spuriously rejects 2--3\,cm same-object drift.

\textbf{Held tracks} (separate branch, \code{rigid\_attachment\_predict}):
$\mu\!\leftarrow\!\Delta T_{bg}\mu$ with $\Delta T_{bg}=T_{bg,k}T_{bg,k-1}^{-1}$,
$P\!\leftarrow\!\Ad{\Delta T_{bg}}P\Ad{\Delta T_{bg}}^{\!\top}+Q_{\text{manip}}$.
Held tracks must be \emph{skipped} from \code{predict\_static}; otherwise
the static $u_k$ composes onto $\Delta T_{bg}\mu$ and the kinematics
$T_{bo}=T_{bg}T_{go}$ break.

$Q$ is selected by
\code{process\_noise\_for\_phase(phase, is\_target, frame, frames\_since\_obs)}:
\begin{itemize}[leftmargin=1.5em,itemsep=0pt,topsep=2pt]
\item \emph{Static} ($\neg$is\_target): $Q$ decays with $\code{frames\_since\_obs}$ from
   $\Norm(0,10^{-6})$ down to $\Norm(0,10^{-8})$ per axis.
\item \emph{Held, holding}: $Q\!=\!\operatorname{diag}(2.5\!\cdot\!10^{-4},10^{-3})$
   per trans / rot axis. Models grip slack + proprioception jitter.
\item \emph{Held, grasping/releasing}: $Q\!=\!I$ (essentially unknown).
   The transition frame's pose is not a function of $\Delta T_{bg}$.
\end{itemize}
Bernoulli predict: $r_{k|k-1}=p_s r_{k-1|k-1}$ (default $p_s=1$).

\subsection{Step 2. Coarse measurement (cleaned mask centroid)}\label{sec:coarse}

For each detection: $c^{co}_l=\code{centroid\_cam\_from\_mask}(m_l,D_k,K)$.
The mask is first passed through \code{\_clean\_mask}
(\code{pose\_update/icp\_pose.py}):
\begin{enumerate}[leftmargin=1.5em,itemsep=0pt,topsep=2pt]
\item Binary erosion (1 pixel) drops the worst boundary band.
\item Per-masked-pixel 4-neighbour depth-jump rejection
   ($|\Delta d|\!>\!3\,\text{cm}$ kills boundary pixels straddling
   foreground/background; object curvature is sub-cm).
\end{enumerate}
Both stages \emph{self-disable} below 30 surviving pixels (small masks
keep their thin rim). On \code{apple\_in\_the\_tray} frame 300 (a
$\sim\!15\text{k}$-pixel tray detection): retains 90\% of pixels,
shrinks \code{obj\_radius} from 19.9 to 16.4\,cm by removing background
leakage tails. Centroid shift is sub-mm (filter is spatially symmetric);
ICP fitness stable.

The Hungarian cost uses isotropic $R_{\text{cent}}=\operatorname{diag}((2\,\text{cm})^2 I_3)$.
Detections that fail back-projection ($<\!30$ valid-depth pixels even
after cleaning) are dropped from association \emph{and} from the birth
buffer.

\subsection{Step 3. Hungarian association}\label{sec:assoc}

\code{association.hungarian\_associate}, called with a 3-DOF centroid
innovation:
\begin{align}
   \nu &= [\nu_{3D},\bm 0]^{\!\top}\in\R^6, \quad
   S = \operatorname{diag}(S_{3D},\,\tau I_3), \\
   d^2_t &= \nu_{3D}^{\!\top}S_{3D}^{-1}\nu_{3D}, \quad d^2_r=0
\end{align}
with $\tau=10^6$ in the rotation slot to make the block decomposition
cleanly 3-DOF. $S_{3D}=P^{(i)}_{tt}+R^{bo}_{\text{cent}}$ where
$R^{bo}_{\text{cent}}=R_{bc}R^{cam}_{\text{cent}}R_{bc}^{\!\top}$ is
the rotation-only base-frame lift of the camera centroid covariance.
A naive 6-DOF $R_{icp}=\operatorname{diag}(R_t,\tau I)$ would couple
$\tau$ into translation via $[t_{bc}]_{\!\times}$ and collapse $d^2_t\to 0$.

\textbf{Cost} (in $\chi^2$ units, \code{cost\_d2\_mode='sum'}):
\begin{equation}
   C_{il} = (d^2_{il,t}+0.1\,d^2_{il,r})
      \;-\;\alpha\,\ind[\tau_l\!=\!\tau^{(i)},\tau^{(i)}\!\ge\!0]
      \;+\;\beta\,\ind[\hat\ell_l\!\notin\!\mathcal H^{(i)}]
      \;+\;\gamma\,\max(0,1-\hat s_l)
\end{equation}
with $\alpha\!=\!4.4$ (SAM2-id continuity), $\beta\!=\!6.0$ (label-flicker
soft penalty, $<G_{\text{out},t}$ so geometry can override label),
$\gamma\!=\!2.0$ (low-score mild penalty). Outer gate
$d^2_{il,t}\le 21.108$ (\code{gate\_mode='trans'}); cells failing it get
the sentinel \code{\_INFEASIBLE}=$10^{12}$.

Solver: \code{scipy.optimize.linear\_sum\_assignment} on the
rectangular cost matrix; assignments at \code{\_INFEASIBLE} are
post-filtered out (Hungarian on a fully-infeasible row will still emit
some column). Result: \code{match\{oid: det\_idx\}}.

\textbf{Recovering the paper baseline}: \code{enforce\_label\_match=True},
$\beta=\gamma=0$, \code{gate\_mode='full'}, \code{cost\_d2\_mode='full'}.
\textbf{Oracle mode}: \code{association\_mode="oracle"} bypasses the
cost matrix and matches by upstream \code{det["id"]}; used by the
degeneracy test.

\subsection{Step 3b. Per-match fine ICP}\label{sec:icp-fine}

For every $(i,l)\in\pi^*$:
\code{PoseEstimator.estimate(oid=i, mask=m\_l, depth=D, T\_co\_init=
\(\hat T^{co}_{k|k-1,i}\))}. The seed is the matched track's filter
prior; ICP refines against \code{\_refs[i]} (\emph{tracker-oid}-keyed,
stable across SAM2 id reissues for the same physical object).

Output $\hat R^{co}$ uses data-driven floors:
\begin{align*}
   \sigma^2_t &= \max(\text{rmse}^2,(2\,\text{cm})^2)\cdot\text{fit\_scale}, \\
   \sigma^2_r &= \max((\text{rmse}/r_{\text{obj}})^2,(0.1\,\text{rad})^2)\cdot\text{fit\_scale}.
\end{align*}
ICP failure (\code{fitness}\,$<\!0.90$ or \code{rmse}\,$>\!15$\,mm) demotes the pair to a centroid-only update with inflated rotation $R$ (no
rotation information through Kalman).

\subsection{Step 4. Visibility \texorpdfstring{$p_v^{(i)}$}{p\_v}}\label{sec:vis}

$p_v^{(i)}=\ind[\pi_K(T_{bc}^{-1}T)\in\Omega]\cdot\prod_{j\ne i}(1-\rho_{ij})$
(\S\ref{B-sec:vis}). Implementation: \code{visibility.visibility\_p\_v}.
Off when \code{enable\_visibility=False}; Production orchestrator still
uses constant $T_{bc}$ here (test driver uses per-frame).

\subsection{Step 5. Matched-track update}\label{sec:match}

For matched $(i,l)$: lift the camera-frame measurement to base via
$T_{bo,\text{meas}}=T_{bc}(t_k)\hat T^{co}$,
$R^{bo}=\Ad{T_{bc}}\hat R^{co}\Ad{T_{bc}}^{\!\top}$. Then IEKF (2 inner
iterations) + Joseph form + $\Phi$ saturate + diagonal floor on $P$.

Existence update: $r\!\leftarrow\!\code{r\_assoc\_update\_loglik}(r,\log\mathcal L,p_d,\lambda_c)$
in log-space (avoids underflow for $\log\mathcal L\!\sim\!-50$).

Robust gate: Huber weight on $d^2_t$ ($G_{\text{in},t}$ inner soft, $G_{\text{out},t}$ outer hard); $w=0$ tracks
revert to the miss branch. Append $(k,\hat T^{co},\hat R^{co})$ to the
loop-closure-aware chain $\mathcal H^{(i)}$ (\S\ref{B-sec:chain}).

\subsection{Step 6. Missed-track update}\label{sec:miss}

For unmatched $i$: $r\!\leftarrow\!(1-\tilde p_d)r/((1-\tilde p_d)r+(1-r))$,
$\tilde p_d=p_d p_v^{(i)}$. $T,P$ unchanged. When $p_v\!\to\!0$ the update collapses to
$r_{k|k}=r_{k|k-1}$.

\subsection{Step 7. Birth via pending buffer}\label{sec:birth}

Two-phase, with the \emph{tracker} owning admission counters:

\textbf{Pending buffer} (\code{self.\_pending\_births}, dict keyed by
perception id). For each unmatched detection $l$ that passes the
back-projection gate:
\begin{enumerate}[leftmargin=1.5em,itemsep=0pt,topsep=2pt]
\item Look up / create \code{\_PendingBirth(source\_id=det["id"], \dots)}; bump
   \code{n\_obs\_tracker}, update \code{max\_score}.
\item \textbf{Policy gate} (\code{birth\_admissible} with \code{require\_pose=False}):
   bbox not on image border ($\ge$2px), \code{n\_obs\_tracker}$\ge K\!=\!3$,
   \code{max\_score}$\ge S\!=\!0.20$. Note: $K$ counts the
   \emph{tracker}'s unmatched-streak frames, not perception's
   \code{n\_obs} (which resets on SAM2 id reissue).
\item If admitted: mint $i_{\text{new}}=\code{\_next\_track\_id()}$ (fresh
   consecutive int, NOT seeded from \code{det["id"]}); call
   \code{PoseEstimator.estimate(oid=i\_new, T\_co\_init=None)} once
   (first-observation branch: anchors at $c^{co}_l$, populates
   \code{\_refs[i\_new]}); commit via \code{\_birth\_track}; pop the
   pending entry. Bernoulli init: $r=\lambda_b\hat s/(\lambda_b\hat s+\lambda_c)$.
\item If rejected: keep the pending entry. TTL-expire after
   \code{birth\_pending\_ttl\_frames}=30 of no re-arrival (one second
   at 30\,Hz; survives short perception dropouts).
\end{enumerate}

This invariant matters: \emph{$i\!\in\!\code{\_refs}$ iff $i$ is a
committed track}. The pre-2026 path called \code{estimate} as an
admission probe, seeding \code{\_refs[i]} before the policy gate and
leaking it on rejection; subsequent frames then ran refine-ICP against
that stale cloud and silently dropped real birth candidates (manifested
as the apple-in-tray + tray failing to ever birth).

Birth covariance: $P_{\text{new}}=\operatorname{diag}((5\,\text{cm})^2,(0.05\,\text{rad})^2)$
(default; \code{init\_cov\_from\_R=True} uses the ICP $\hat R$ instead).

\subsection{Step 8. Prune + self-merge}

\textbf{Prune} (\code{\_prune\_track}): drop $i$ with $r<r_{\min}=10^{-3}$
from \code{state.objects}, all per-orchestrator dicts, and \code{\_refs[i]}.

\textbf{Self-merge} (\code{\_self\_merge\_pass(held\_id)}): Hungarian's
one-to-one constraint leaves duplicates when SAM2 emits parallel ids for
one physical object. Per-frame greedy fusion of same-label pairs with
$\|\mu_i\!-\!\mu_j\|\le \code{self\_merge\_trans\_m}=5\,\text{cm}$. Keep
decision: held oid wins (preserves rigid-attachment kinematics);
otherwise longer chain wins; tie-break lower oid. Fusion is Bayesian
information sum on $\SE$ linearised at $\mu_{\text{keep}}$:
\begin{equation}
   P^{-1}_{\text{new}} = P^{-1}_{\text{keep}}+P^{-1}_{\text{drop}}, \quad
   \mu_{\text{new}} = \mu_{\text{keep}}\,
      \Exp\!\bigl(P_{\text{new}}P^{-1}_{\text{drop}}\Log(\mu_{\text{keep}}^{-1}\mu_{\text{drop}})\bigr).
\end{equation}
Bookkeeping: $r\!\leftarrow\!\max$, \code{frames\_since\_obs}$\leftarrow\!\min$,
\code{label\_scores} per-label sum + sample-weighted mean,
$\mathcal H^{(\text{drop})}$ appended to $\mathcal H^{(\text{keep})}$,
\code{\_refs[drop]} merged into \code{\_refs[keep]} via voxel-downsample.

\subsection{Step 9. Emit}

\code{TwoTierOrchestrator.objects} returns
$\{i:r^{(i)}\!\ge\!r_{\text{conf}}\!=\!0.5\}$. Tentative tracks are still
predicted/associated/updated but hidden; visible via
\code{tentative\_objects}. World-frame composition $T_{wo}\!=\!T_{wb}T_{bo}$
(or the chain smoother of \S\ref{B-sec:chain} when bundle-adjusted).

\section{Held-object handling}\label{sec:held}
%-----------------------------------------------------------------------------

The dataset has no annotated \code{phase}; the test driver's
\code{\_GripperStateInferrer} (\code{tests/visualize\_ekf\_tracking.py})
infers $\{\text{phase},\text{held\_obj\_id}\}$ from proprioception:

\paragraph{Phase from gripper finger width.}
$w\!=\!\text{l\_gripper\_finger\_joint}+\text{r\_gripper\_finger\_joint}$ (from
\code{joints\_pose.json}). Hysteresis bands prevent chatter:
\begin{equation*}
   \text{is\_closed}_k =
   \begin{cases}
      w_k < 0.040\,\text{m}, & \text{previously closed}, \\
      w_k < 0.025\,\text{m}, & \text{previously open}.
   \end{cases}
\end{equation*}
Phase transitions: $\{\text{open}\!\to\!\text{closed}\}=$\,\emph{grasping},
$\{\text{closed}\!\to\!\text{closed}\}=$\,\emph{holding},
$\{\text{closed}\!\to\!\text{open}\}=$\,\emph{releasing},
otherwise \emph{idle}.

\paragraph{Held identity from EE proximity.}
At grasp onset (idle$\to$grasping) compute the world-frame EE position
$T_{we}\!=\!T_{wb}T_{bg}$ and pick the nearest live track within
\code{grasp\_radius\_m}=30\,cm; that becomes \code{held\_obj\_id}.
The id is \emph{frozen} through holding (no per-frame re-inference --
that would jitter as masks resegment), released at the
\emph{releasing} edge, and migrated through self-merges via
\code{apply\_merges} (which reassigns to the keeper if the held oid
was a drop).

\paragraph{Q schedule.}
The orchestrator passes $\{T_{bg}, \text{held\_oids}, \text{phase}\}$ to
\code{tracker.step()}. \code{rigid\_attachment\_predict} reads the
phase-dependent $Q$ from \code{process\_noise\_for\_phase(phase,
is\_target=True, frame="base")}: huge during grasping/releasing
(transition pose is unknown), $\sim\!1.6\,\text{cm}/1.8^\circ$ per frame
during holding (grip slack + proprio jitter). The pre-fix
$\Norm(0,10^{-8})$ value was unrealistically tight; $P$ stayed pinned
at the floor and the Hungarian gate rejected every re-detection of
the moving held object.

\paragraph{Self-merge protection.}
\code{\_self\_merge\_pass(held\_id=\dots)} forces the held oid as
\code{keep} in any pair it participates in, preserving the
rigid-attachment motion model across merges.

\section{Known deviations from \code{bernoulli\_ekf.tex}}
%-----------------------------------------------------------------------------

\begin{enumerate}[leftmargin=2em,itemsep=0pt,topsep=2pt]
\item Standalone \code{ekf\_se3.ekf\_update} uses $(I-K)P$ rather than
   Joseph and skips $\Phi$. The RBPF Bernoulli path
   (\code{RBPFState.update\_observation}) is paper-correct.
\item Production orchestrator still passes constant $T_{bc}$ to the
   visibility predicate; the test driver installs per-frame
   $T_{bc}(t_k)$ via \code{set\_camera\_extrinsic}. Migrate by adding
   the call in \code{step()} before any measurement-side function;
   until then, set \code{enable\_visibility=False} on the production
   path or accept head-pan-aliased visibility flicker.
\item Visibility footprint $A^{(i)}$ uses the matched-detection bbox as
   a proxy. Paper prescribes projecting the TSDF shape summary
   $\mathcal G^{(i)}$.
\item \emph{Mask-edge cloud filter} (\S\ref{sec:coarse}) and
   \emph{rigid-attachment predict with phase-dependent $Q$}
   (\S\ref{sec:held}) are manipulation-aware extensions not in the
   paper.
\item \emph{Pending-birth buffer} (\S\ref{sec:birth}) replaces the
   paper's per-frame ``unmatched detection $\to$ immediate birth''
   with a tracker-side admission policy. The Bernoulli existence
   $r$ semantics are unchanged; only the gating before $r_{\text{new}}$
   is computed differs.
\end{enumerate}
